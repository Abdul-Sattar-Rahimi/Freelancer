source = close


rsiMA_barColor(isExp, rsi_src, length1, length2) =>
    ma1 = isExp ? ema(rsi_src, length1) : sma(rsi_src, length1)
    ma2 = isExp ? ema(rsi_src, length2) : sma(rsi_src, length2)
    ma1Change = change(ma1)
    output = ma1Change >= 0 and ma1>ma2 ? lime :
        ma1Change < 0  and ma1>ma2 ? red :
        ma1Change <= 0 and ma1<ma2 ? maroon :
        ma1Change >= 0 and ma1<ma2 ? green : gray
    output
